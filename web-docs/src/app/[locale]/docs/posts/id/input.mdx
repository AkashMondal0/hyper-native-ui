## Props

### `lightColor`
- **Type**: `string`
- **Optional**
- **Description**: Custom background color for light mode.

### `darkColor`
- **Type**: `string`
- **Optional**
- **Description**: Custom background color for dark mode.

### `disabled`
- **Type**: `boolean`
- **Optional**
- **Description**: Disables the input field if set to `true`.

### `themeScheme`
- **Type**: `"light" | "dark"`
- **Optional**
- **Description**: Overrides the current theme scheme.

### `rightSideComponent`
- **Type**: `React.ReactNode`
- **Optional**
- **Description**: Component displayed on the right side of the input field.

### `leftSideComponent`
- **Type**: `React.ReactNode`
- **Optional**
- **Description**: Component displayed on the left side of the input field.

### `onFocus`
- **Type**: `() => void`
- **Optional**
- **Description**: Callback triggered when the input gains focus.

### `onBlur`
- **Type**: `() => void`
- **Optional**
- **Description**: Callback triggered when the input loses focus.

### `isErrorBorder`
- **Type**: `boolean | unknown`
- **Optional**
- **Description**: Displays an error border if set to `true`.

### `placeholder`
- **Type**: `string`
- **Optional**
- **Default**: `'Enter text here'`
- **Description**: Placeholder text displayed when the input is empty.

### `containerStyle`
- **Type**: `ViewProps["style"]`
- **Optional**
- **Description**: Style object for the container view.

### `variant`
- **Type**: `"default" | "secondary" | ThemeName`
- **Optional**
- **Default**: `"default"`
- **Description**: Determines the visual style of the input field. Variants include custom theme names.

### `size`
- **Type**: `"small" | "medium" | "large" | "extraLarge"`
- **Optional**
- **Default**: `"medium"`
- **Description**: Specifies the size of the input field.

### `...props`
- **Type**: `TextInputProps`
- **Optional**
- **Description**: Additional props passed to the underlying `TextInput` component.

## Behavior

The `Input` component adjusts its appearance based on the provided `variant`, `size`, and `themeScheme`. It uses the `useTheme` hook to dynamically apply theme colors and styles.

### Theme Color Resolution Logic
1. If the `variant` is `"default"` or `"secondary"`, current theme colors are used.
2. For custom variants (e.g., `"Red"`, `"Orange"`), matching colors from `themeColors` are applied.
3. Fallback colors are used when no matching theme or variant is found.

### Size Variants
- **`small`**: Smaller font size for compact inputs.
- **`medium`**: Default size.
- **`large`**: Larger font size for more prominent inputs.
- **`extraLarge`**: Extra-large font size for high-visibility inputs.

## Component Code
<CodeBlock className="my-24"
    compact
    codeInstances={[
        {
            code: `import React from 'react';
import { type TextInputProps, type ViewProps, TextInput, View } from 'react-native';
import { themeColors } from '../constants/Colors';
import useTheme from '../hooks/useTheme';

export type ThemeName = "Zinc" | "State" | "Stone" | "Grey" | "Neutral" | "Red" | "Rose" | "Orange" | "Yellow" | "Green";
export type Props = TextInputProps & {
    lightColor?: string;
    darkColor?: string;
    disabled?: boolean;
    themeScheme?: "light" | "dark";
    rightSideComponent?: React.ReactNode;
    leftSideComponent?: React.ReactNode;
    onFocus?: () => void;
    onBlur?: () => void;
    isErrorBorder?: boolean | unknown
    placeholder?: string
    containerStyle?: ViewProps["style"]
    variant?: "default" | "secondary" | ThemeName
    size?: "small" | "medium" | "large" | "extraLarge";
};

const Input = ({ disabled,
    onBlur,
    onFocus,
    isErrorBorder,
    themeScheme,
    size = "medium",
    placeholder = 'Enter text here',
    style,
    variant = "default",
    containerStyle,
    ...props }: Props) => {
    const { currentTheme, themeScheme: defaultThemeScheme } = useTheme();
    const [isFocused, setIsFocused] = React.useState(false);

    const colorVariant = () => {
        switch (variant) {
            case "secondary":
                return {
                    backgroundColor: currentTheme.background,
                    color: currentTheme.foreground,
                    borderColor: currentTheme.border,
                    isFocused: currentTheme.primary
                };
            case "default":
                return {
                    backgroundColor: currentTheme.input,
                    color: currentTheme.foreground,
                    borderColor: currentTheme.border,
                    isFocused: currentTheme.primary
                };
            default:
                const theme = themeColors.find((theme) => theme.name === variant)![themeScheme ?? defaultThemeScheme];
                if (theme) {
                    return {
                        backgroundColor: theme.input,
                        color: theme.foreground,
                        borderColor: theme.border,
                        isFocused: theme.primary
                    };
                }
                return {
                    backgroundColor: currentTheme.input,
                    color: currentTheme.foreground,
                    borderColor: currentTheme.border,
                    isFocused: currentTheme.primary
                };
        }
    };

    const sizeVariant = () => {
        switch (size) {
            case "small":
                return {
                    fontSize: 12,
                };
            case "large":
                return {
                    fontSize: 22
                };
            case "extraLarge":
                return {
                    fontSize: 28
                };
            default:
                return {
                    fontSize: 16
                };
        }
    };

    const inputStyle = {
        ...colorVariant(),
        ...sizeVariant()
    };

    if (!currentTheme) return <></>;

    return (
        <View
            style={[{
                flexDirection: 'row',
                alignItems: 'center',
                borderRadius: 14,
                borderWidth: 1,
                paddingHorizontal: 4,
                opacity: disabled ? 0.5 : 1,
                borderColor: isErrorBorder ? currentTheme?.destructive : disabled || !isFocused ? inputStyle?.borderColor : inputStyle?.isFocused,
                backgroundColor: inputStyle.backgroundColor,
            }, containerStyle]}>
            {props.leftSideComponent}
            <TextInput
                style={[{
                    color: inputStyle.color,
                    height: "auto",
                    width: '100%',
                    padding: 6,
                    fontSize: inputStyle.fontSize
                }, style]}
                selectionHandleColor={currentTheme?.primary}
                placeholderTextColor={currentTheme?.muted_foreground}
                onFocus={() => {
                    onFocus && onFocus();
                    setIsFocused(true);
                }}
                onBlur={() => {
                    onBlur && onBlur();
                    setIsFocused(false);
                }}
                placeholder={placeholder}
                editable={!disabled}
                {...props} />
            {props.rightSideComponent}
        </View>
    );
};

export default Input;`,
            label: 'React Native',
            language: 'tsx'
        },
    ]}
    copyButton
/>
